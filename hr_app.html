function doGet(e) { var action = e.parameter.action; var callback = e.parameter.callback; var result = { success: false }; try { var sheet = SpreadsheetApp.getActiveSpreadsheet(); var maNV = String(e.parameter.maNV || '').trim().toUpperCase(); // 1. TÍNH NĂNG ĐĂNG NHẬP if (action === 'login') { var matKhau = String(e.parameter.matKhau || '').trim(); var ws = sheet.getSheetByName('NhanVien') || sheet.getSheets()[0]; var data = ws.getDataRange().getValues(); for (var i = 1; i < data.length; i++) { if (String(data[i][0]).trim().toUpperCase() === maNV && String(data[i][6]).trim() === matKhau) { result = { success: true, maNV: String(data[i][0]), hoTen: String(data[i][1]), phongBan: String(data[i][3]), chucVu: String(data[i][4]) }; break; } } } // 2. TÍNH NĂNG LẤY NGÀY PHÉP else if (action === 'getLeave') { var ws = sheet.getSheetByName('NhanVien') || sheet.getSheets()[0]; var data = ws.getDataRange().getValues(); for (var i = 1; i < data.length; i++) { if (String(data[i][0]).trim().toUpperCase() === maNV) { result = { success: true, phepNamTong: Number(data[i][11]) || 12, // Cột L phepNamDaDung: Number(data[i][12]) || 0, // Cột M phepNamConLai: Number(data[i][13]) || 0, // Cột N nghiBuConLai: Number(data[i][14]) || 0 // Cột O }; break; } } } // 3. TÍNH NĂNG LẤY LỊCH LÀM VIỆC else if (action === 'getSchedule') { var ws = sheet.getSheetByName('LichLamViec') || sheet.getSheetByName('Lịch làm việc') || sheet.getSheets()[1]; if (ws) { var data = ws.getDataRange().getValues(); var headers = data[0]; var userRow = null; for (var i = 1; i < data.length; i++) { if (String(data[i][0]).trim().toUpperCase() === maNV) { userRow = data[i]; break; } } var list = []; if (userRow) { for (var j = 1; j < headers.length; j++) { if (headers[j]) { var dStr = headers[j] instanceof Date ? Utilities.formatDate(headers[j], Session.getScriptTimeZone(), "dd/MM/yyyy") : String(headers[j]); list.push({ date: dStr, ca: String(userRow[j] || '—').trim() }); } } } result = { success: true, schedule: list }; } } // 4. TÍNH NĂNG LẤY YÊU CẦU XIN NGHỈ else if (action === 'getRequests') { result = { success: true, requests: [] }; } } catch(err) { result.error = err.toString(); } var memo = callback + "(" + JSON.stringify(result) + ");"; return ContentService.createTextOutput(memo).setMimeType(ContentService.MimeType.JAVASCRIPT); }
